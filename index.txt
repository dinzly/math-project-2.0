<script>
    // Generate simulated data points matching typical Malaysian SAGA biological metrics over 42 days
    const labels = [];
    const weightData = [];
    const rateData = [];

    // Cubic weight formula: f(x) = -0.022x^3 + 1.62x^2 + 2.5x + 45
    function calculateWeight(x) {
        return (-0.022 * Math.pow(x, 3)) + (1.62 * Math.pow(x, 2)) + (2.5 * x) + 45;
    }

    // Derivative growth velocity formula: f'(x) = -0.066x^2 + 3.24x + 2.5
    function calculateGrowthRate(x) {
        return (-0.066 * Math.pow(x, 2)) + (3.24 * x) + 2.5;
    }

    // Populate data sets at 3-day intervals for balanced visualization tracking
    for (let day = 0; day <= 42; day += 3) {
        labels.push("Day " + day);
        weightData.push(parseFloat(calculateWeight(day).toFixed(2)));
        rateData.push(parseFloat(calculateGrowthRate(day).toFixed(2)));
    }

    // Initialize Chart.js configuration
    const ctx = document.getElementById('growthChart').getContext('2d');
    const growthChart = new Chart(ctx, {
        type: 'line',
        data: {
            labels: labels,
            datasets: [{
                label: 'SAGA Body Mass Weight f(x) [Grams]',
                data: weightData,
                borderColor: '#2e7d32',
                backgroundColor: 'rgba(46, 125, 50, 0.1)',
                borderWidth: 3,
                tension: 0.2,
                yAxisID: 'yWeight',
                fill: true,
                pointRadius: 5,
                pointHoverRadius: 8,
                pointBackgroundColor: '#2e7d32'
            }]
        },
        options: {
            responsive: true,
            maintainAspectRatio: true,
            plugins: {
                legend: {
                    position: 'top',
                },
                tooltip: {
                    callbacks: {
                        label: function(context) {
                            return context.dataset.label + ": " + context.raw + " g";
                        }
                    }
                }
            },
            scales: {
                yWeight: {
                    type: 'linear',
                    position: 'left',
                    title: {
                        display: true,
                        text: 'Cumulative Bird Mass (g)',
                        color: '#2e7d32',
                        font: { weight: 'bold' }
                    },
                    min: 0,
                    max: 2500
                },
                x: {
                    title: {
                        display: true,
                        text: 'Broiler Production Age Timeline (Days)',
                        font: { weight: 'bold' }
                    }
                }
            },
            // Logic targeting element intersections on mouse clicks
            onClick: (event, activeElements) => {
                if (activeElements.length > 0) {
                    const dataIndex = activeElements[0].index;
                    const selectedDay = dataIndex * 3; // Corresponds to the 3-day interval logic
                    
                    updateDiagnosticPanel(selectedDay);
                }
            }
        }
    });

    // Handle dynamic calculus calculation panel rendering updates
    function updateDiagnosticPanel(day) {
        const weight = calculateWeight(day);
        const rate = calculateGrowthRate(day);
        
        const badgeEl = document.getElementById('dayBadge');
        const contentEl = document.getElementById('panelContent');
        const panelEl = document.getElementById('outputPanel');

        badgeEl.innerText = "Day " + day + " Analysis";
        panelEl.style.backgroundColor = "#fffde7"; // Briefly shift color to cue layout rendering changes
        setTimeout(() => { panelEl.style.backgroundColor = "var(--primary-light)"; }, 150);

        // Generate specific interpretations mapped directly to broiler physiology milestones
        let interpretation = "";
        if (day <= 12) {
            interpretation = "<strong>Broiler Brooding Stage:</strong> The tissue accumulation velocity is small but increasing. Energy usage goes toward establishing skeletal systems, internal organs, and metabolic homeostasis. Keep environmental housing temperatures tightly regulated between 29-32°C inside closed houses.";
        } else if (day > 12 && day <= 27) {
            interpretation = "<strong>Peak Dynamic Acceleration Stage:</strong> The derivative value shows the flock is gaining muscle tissue rapidly. This phase requires optimized feed profiles with balanced amino acids to maximize growth performance and support protein deposition.";
        } else if (day > 27 && day <= 36) {
            interpretation = "<strong>Maturation Inflection Phase:</strong> Although the overall weight curve continues ascending, the derivative confirms the growth rate has flattened out and begun a downward trend. Protein accretion shifts toward abdominal fat pad production.";
        } else {
            interpretation = "<strong>Harvest Target Window:</strong> The growth rate slows down significantly. Keeping birds past this timeline reduces feed conversion efficiency and cuts into profit margins. Plan harvest schedules and coordinate logistics with processing facilities.";
        }

        // Structural HTML generation inserted into the application UI layout
        contentEl.innerHTML = `
            <table style="width:100%; border-collapse: collapse; margin-bottom: 15px;">
                <tr>
                    <td style="padding: 6px 0; color:#555;"><strong>Current Age Position ($x$):</strong></td>
                    <td style="text-align: right; font-weight: bold;">${day} Days Old</td>
                </tr>
                <tr>
                    <td style="padding: 6px 0; color:#555;"><strong>Flock Mass $f(x)$:</strong></td>
                    <td style="text-align: right; color: var(--primary); font-weight: bold;">${weight.toFixed(2)} Grams</td>
                </tr>
                <tr>
                    <td style="padding: 6px 0; color:#555;"><strong>Instantaneous Derivative Value $f'(x)$:</strong></td>
                    <td style="text-align: right; color: var(--secondary); font-weight: bold;">${rate.toFixed(2)} Grams / Day</td>
                </tr>
            </table>
            <div style="background: white; padding: 12px; border-radius: 6px; border: 1px solid #d4edda; font-size: 14px; line-height: 1.5;">
                <strong>Technologist Diagnostic Note:</strong> ${interpretation}
            </div>
        `;
    }
</script>
